  9    """
 10    Function that splits the elements of t between two arrays.
 11
 12    Args:
 13        - t: array to be split.
 14
 15    Returns:
 16        - Tuple[np.ndarray, int, np.ndarray] with:
 17            - numpy array with elements smaller than the pivot.
 18            - integer number used to split the array (the pivot).
 19            - numpy array with elements greater than the pivot.

 22    if t is None:
 23        print("Error in given array.")
 24        return None
 25
 26    # get pivot

 28    if len(t) > 0:

 35        if x < pivot:
 36            l.append(x)

 38            r.append(x)

 45def qsel(t: np.ndarray, k: int) -> Union[int, None]:
 46    """
 47    Function that implements the recursive QuickSelect algorithm using the split function.
 48
 49    Args:
 50        - t: array with elements.
 51        - k: index to search in the array.
 52
 53    Returns:
 54        - integer representing the element that would occupy the k position of ordering t (the list)
 55        or None if this element does not exist (because the index is greater or smaller than the ones in the table).
 56    """
 57
 58    # Error Control
 59    if t is None or k < 0 or k > len(t):
 60        return None
 61
 62    # array with 1 element
 63    if len(t) == 1 and k == 0:
 64        return t[0]
 65
 66    # array with >1 element
 67    else:
 68        (left, pivot, right) = split(t)
 69        if pivot == -1:
 70            return None
 71
 72        # m = index of pivot
 73        m = len(left)
 74
 75        # if k is same as m then i found the element
 76        if k == m:
 77            return pivot
 78        elif k < m:
 79            # no need to adjust k since it is the left subarray
 80            return qsel(left, k)
 81        elif k > m:
 82            # adjust new k to the right subarray
 83            return qsel(right, k - m - 1)

 87    """
 88    Function that implements QuickSelect algorithm with Split function and without recursive calls.

 92        - k: index to search in the array.
 93
 94    Returns:
 95        - integer representing the element that would occupy the k position of ordering t (the list)
 96        or None if this element does not exist (because the index is greater or smaller than the table indexes).
 97    """

104        return t[0]
105
106    # Array with >1 element
107    m = -1

163def pivot5(t: np.ndarray) -> int:
164    """
165    Function returns the ”pivot5” of the array t accordingt to the procedure ”median of the medians of 5 elements”.
166
167    Args:
168        - t: array with elements.
169
170    Returns:
171        - integer representing the pivot5.
172    """
173    # Error Control
174    if t is None:
175        return None
176
177    # Median Of Medians: mom
178    if len(t) < 5:
179        return sorted(t)[int(len(t) / 2)]
180
181    mom = []
182    i = 0
183
184    while i < len(t):
185        sub = len(t) - i
186        if sub < 5:
187            sub5 = t[i:]
188            five = sorted(sub5)[int(len(sub5) / 2)]
189            mom.append(five)
190            break
191        else:
192            sub5 = t[i:i + 4]
193            five = sorted(sub5)[int(len(sub5) / 2)]
194            mom.append(five)
195            i += 5
196
197    pivot = qsel5_nr(mom, int(len(mom) / 2))
198    return pivot

201def qsel5_nr(t: np.ndarray, k: int) -> Union[int, None]:
202    """
203    Function that implements QuickSelect5 algorithm with split_pivot function and without recursive calls.
204
205    Args:
206        - t: array with elements.
207        - k: index to search in the array.
208
209    Returns:
210        - integer representing the element that would occupy the k position of ordering t
211        or None if this element does not exist.
212    """
213
214    # Error Control
215    if k >= len(t) or k < 0 or t is None or len(t) < 0:
216        return None
217
218    # Array with 1 element
219    if len(t) == 1 and k == 0:
220        return t[0]
221
222    # Array with >1 element
223    m = -1
224    pivot = -1
225    while k != len(t):
226        # get pivot
227        pivot = pivot5(t)
228        if pivot == -1:
229            return None
230        else:
231            # partition
232            (left, pivot, right) = split_pivot(t, pivot)
233            # m = index of pivot
234            m = len(left)
235            if k == m:
236                break
237            elif k < m:
238                t = left
239            elif k > m:
240                t = right
241                k = k - m - 1
242    return pivot

251    Args:

254    Returns:
255        - sorted numpy array.
256    """
257    # Error Control
258    if t is None or len(t) < 0:
259        return None

280def edit_distance(str_1: str, str_2: str) -> int:
281    """
282    Function that returns the edit distance between the two strings.
283
284    Args:
285        - str_1: first string.
286        - str_2: second string.
287
288    Returns:
289        - integer representing the edit distance between the two strings.
290    """
291    # Error Control
292    if str_1 is None or str_2 is None:
293        return None
294
295    # lengths of strings
296    m = len(str_1)
297    n = len(str_2)
298
299    # To store our results
300    d = np.zeros((m + 1, n + 1), int)
301
302    for i in range(m + 1):
303        for j in range(n + 1):
304            # CASE 1: str_1 is empty -> INSERT second string
305            if i == 0:
306                d[i][j] = j  # min=j
307
308            # CASE 2: str_2 is empty -> INSERT first string
309            elif j == 0:
310                d[i][j] = i  # min=i
311
312            # CASE 3: final chars are equal -> ignore last char
313            elif str_1[i - 1] == str_2[j - 1]:
314                d[i][j] = d[i - 1][j - 1]
315
316            # CASE 4: rest of possibilities -> find min
317            else:
318                # --------------/--INSERT--/--DELETE--/--DELETE+INSERT--/
319                d[i][j] = 1 + min(d[i][j - 1], d[i - 1][j], d[i - 1][j - 1])
320
321    return d[m][n]

324def max_subsequence_length(str_1: str, str_2: str) -> int:
325    """
326    Function that returns the length of the longest subsequence common both strings, even is not contiguous
327    and using as little memory as possible.
328
329    Args:
330        - str_1: first string.
331        - str_2: second string.
332
333    Returns:
334        - integer representing the length.
335    """
336    # Error Control
337    if str_1 is None or str_2 is None:
338        return None
339
340    # lengths of strings
341    m = len(str_1)
342    n = len(str_2)
343
344    if m < 0 or n < 0:
345        return None
346
347    # To store our results
348    d = np.zeros((m + 1, n + 1), int)
349
350    for i in range(1, m + 1):
351        for j in range(1, n + 1):
352            if str_1[i - 1] == str_2[j - 1]:
353                d[i][j] = 1 + d[i - 1][j - 1]
354            else:
355                d[i][j] = max(d[i - 1][j], d[i][j - 1])
356
357    return d[m][n]

360def max_common_subsequence(str_1: str, str_2: str) -> str:
361    """
362    Function that returns the longest subsequence common to both strings, even is not contiguous.
363
364    Args:
365        - str_1: first string.
366        - str_2: second string.
367
368    Returns:
369        - string representing the subsequence.
370    """
371
372    # Error Control
373    if str_1 is None or str_2 is None:
374        return None
375
376    # lengths of strings
377    m = len(str_1)
378    n = len(str_2)
379
380    if m < 0 or n < 0:
381        return None
382
383    ss = ""
384    d = np.zeros((m + 1, n + 1), int)
385
386    while m > 0 and n > 0:
387        for i in range(1, m + 1):
388            for j in range(1, n + 1):
389                if str_1[i - 1] == str_2[j - 1]:
390                    d[i][j] = 1 + d[i - 1][j - 1]
391                else:
392                    d[i][j] = max(d[i - 1][j], d[i][j - 1])
393                if str_1[m - 1] == str_2[n - 1]:
394                    ss += str_1[m - 1]
395                    m -= 1
396                    n -= 1
397                elif d[m - 1][n] > d[m][n - 1]:
398                    m -= 1
399                else:
400                    n -= 1
401
402    return ss[::-1]

407def min_mult_matrix(l_dims: List[int]) -> int:
408    """
409    Function that returns the minimal number of multiplications necessary to multiply n matrices whose dimensions are contained
410    in the given list.
411
412    Args:
413        - l_dims: list with dimensions (n+1 elements). The first element is the number of rows of the first matrix, and the others
414        are the number of columns of the n matrices.
415
416    Returns:
417        - integer representing the number of multiplications.
418    """
419    n = len(l_dims)
420    mul = np.ones((n, n)) * np.inf
421    mul[0, 0] = 0
422
423    for x in range(n):
424        mul[x][x] = 0
425
426    for i in range(1, n):
427        for j in range(1, n - i + 1):
428            z = j + i - 1
429            for k in range(j, z):
430                c = mul[j][k] + mul[k + 1][z] + \
431                    l_dims[j - 1] * l_dims[k] * l_dims[z]
432                if c < mul[j][z]:
433                    mul[j][z] = c
434
435    return mul[1][n - 1]
